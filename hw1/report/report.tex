\section{Model Architecture}
%==============================================================================

\subsection{Network Structure}
The three-layer fully connected neural network architecture is defined as follows:

\begin{equation}
\begin{aligned}
\mathbf{z}^{(1)} &= \mathbf{W}^{(1)} \mathbf{x} + \mathbf{b}^{(1)} \\
\mathbf{a}^{(1)} &= \sigma(\mathbf{z}^{(1)}) \\
\mathbf{z}^{(2)} &= \mathbf{W}^{(2)} \mathbf{a}^{(1)} + \mathbf{b}^{(2)} \\
\mathbf{a}^{(2)} &= \sigma(\mathbf{z}^{(2)}) \\
\mathbf{z}^{(3)} &= \mathbf{W}^{(3)} \mathbf{a}^{(2)} + \mathbf{b}^{(3)} \\
\hat{\mathbf{y}} &= \text{softmax}(\mathbf{z}^{(3)})
\end{aligned}
\end{equation}

Where:
\begin{itemize}
    \item $\mathbf{x} \in \mathbb{R}^{12288}$: Input vector (flattened image)
    \item $\mathbf{W}^{(1)} \in \mathbb{R}^{12288 \times H}$, $\mathbf{b}^{(1)} \in \mathbb{R}^{H}$: First layer weights and bias
    \item $\mathbf{W}^{(2)} \in \mathbb{R}^{H \times H}$, $\mathbf{b}^{(2)} \in \mathbb{R}^{H}$: Second layer weights and bias
    \item $\mathbf{W}^{(3)} \in \mathbb{R}^{H \times 10}$, $\mathbf{b}^{(3)} \in \mathbb{R}^{10}$: Output layer weights and bias
    \item $H$: Hidden layer size (tunable hyperparameter)
    \item $\sigma$: Activation function (supports ReLU, Sigmoid, Tanh)
\end{itemize}

\subsection{Activation Functions}

\subsubsection{ReLU (Rectified Linear Unit)}
\begin{equation}
\text{ReLU}(x) = \max(0, x)
\end{equation}

Derivative:
\begin{equation}
\text{ReLU}'(x) = \begin{cases} 1, & x > 0 \\ 0, & x \leq 0 \end{cases}
\end{equation}

\subsubsection{Sigmoid}
\begin{equation}
\text{Sigmoid}(x) = \frac{1}{1 + e^{-x}}
\end{equation}

Derivative:
\begin{equation}
\text{Sigmoid}'(x) = \sigma(x)(1 - \sigma(x))
\end{equation}

\subsubsection{Tanh}
\begin{equation}
\text{Tanh}(x) = \frac{e^x - e^{-x}}{e^x + e^{-x}}
\end{equation}

Derivative:
\begin{equation}
\text{Tanh}'(x) = 1 - \tanh^2(x)
\end{equation}

\subsubsection{Softmax}
\begin{equation}
\text{Softmax}(x_i) = \frac{e^{x_i}}{\sum_{j=1}^{C} e^{x_j}}
\end{equation}

\subsection{Loss Function}
The cross-entropy loss with L2 regularization is used:

\begin{equation}
\mathcal{L} = -\frac{1}{N}\sum_{i=1}^{N}\sum_{c=1}^{C} y_{ic} \log(\hat{y}_{ic}) + \frac{\lambda}{2}\sum_{l=1}^{3} \|\mathbf{W}^{(l)}\|_F^2
\end{equation}

Where $N$ is the batch size, $C=10$ is the number of classes, and $\lambda$ is the weight decay coefficient.

\subsection{Weight Initialization}
He initialization (for ReLU) or Xavier initialization (for Sigmoid/Tanh) is used:

\begin{equation}
W_{ij} \sim \mathcal{N}\left(0, \sqrt{\frac{2}{n_{in}}}\right) \quad \text{(He Initialization)}
\end{equation}

\begin{equation}
W_{ij} \sim \mathcal{N}\left(0, \sqrt{\frac{1}{n_{in}}}\right) \quad \text{(Xavier Initialization)}
\end{equation}

\section{Backpropagation Algorithm}
%==============================================================================

\subsection{Forward Propagation}
\begin{algorithm}[H]
\caption{Forward Propagation}
\begin{algorithmic}[1]
\REQUIRE Input $\mathbf{X} \in \mathbb{R}^{N \times 12288}$
\ENSURE Output probabilities $\hat{\mathbf{Y}} \in \mathbb{R}^{N \times 10}$
\STATE $\mathbf{Z}^{(1)} = \mathbf{X}\mathbf{W}^{(1)} + \mathbf{b}^{(1)}$
\STATE $\mathbf{A}^{(1)} = \sigma(\mathbf{Z}^{(1)})$
\STATE $\mathbf{Z}^{(2)} = \mathbf{A}^{(1)}\mathbf{W}^{(2)} + \mathbf{b}^{(2)}$
\STATE $\mathbf{A}^{(2)} = \sigma(\mathbf{Z}^{(2)})$
\STATE $\mathbf{Z}^{(3)} = \mathbf{A}^{(2)}\mathbf{W}^{(3)} + \mathbf{b}^{(3)}$
\STATE $\hat{\mathbf{Y}} = \text{softmax}(\mathbf{Z}^{(3)})$
\RETURN $\hat{\mathbf{Y}}$
\end{algorithmic}
\end{algorithm}

\subsection{Backward Propagation}
\begin{algorithm}[H]
\caption{Backward Propagation}
\begin{algorithmic}[1]
\REQUIRE Predictions $\hat{\mathbf{Y}}$, Ground truth $\mathbf{Y}$
\ENSURE Gradients $\nabla\mathbf{W}^{(l)}, \nabla\mathbf{b}^{(l)}$
\STATE $\delta^{(3)} = \hat{\mathbf{Y}} - \mathbf{Y}$ \COMMENT{Output layer error}
\STATE $\nabla\mathbf{W}^{(3)} = (\mathbf{A}^{(2)})^T \delta^{(3)} / N$
\STATE $\nabla\mathbf{b}^{(3)} = \sum_{i}\delta^{(3)}_i / N$
\STATE $\delta^{(2)} = \delta^{(3)}(\mathbf{W}^{(3)})^T \odot \sigma'(\mathbf{Z}^{(2)})$
\STATE $\nabla\mathbf{W}^{(2)} = (\mathbf{A}^{(1)})^T \delta^{(2)} / N$
\STATE $\nabla\mathbf{b}^{(2)} = \sum_{i}\delta^{(2)}_i / N$
\STATE $\delta^{(1)} = \delta^{(2)}(\mathbf{W}^{(2)})^T \odot \sigma'(\mathbf{Z}^{(1)})$
\STATE $\nabla\mathbf{W}^{(1)} = \mathbf{X}^T \delta^{(1)} / N$
\STATE $\nabla\mathbf{b}^{(1)} = \sum_{i}\delta^{(1)}_i / N$
\RETURN All gradients
\end{algorithmic}
\end{algorithm}

\section{Optimization}
%==============================================================================

\subsection{SGD Optimizer with Momentum}
Stochastic Gradient Descent with momentum is used for optimization:

\begin{equation}
\begin{aligned}
\mathbf{v}_t &= \beta \mathbf{v}_{t-1} - \eta \nabla\mathcal{L} \\
\mathbf{\theta}_t &= \mathbf{\theta}_{t-1} + \mathbf{v}_t
\end{aligned}
\end{equation}

Where $\eta$ is the learning rate and $\beta$ is the momentum coefficient (default: 0.9).

\subsection{Learning Rate Decay}
Multiple learning rate decay strategies are supported:

\subsubsection{Step Decay}
\begin{equation}
\eta_t = \eta_0 \cdot \gamma^{\lfloor t / T \rfloor}
\end{equation}

Where $\gamma$ is the decay rate and $T$ is the decay period.

\subsubsection{Exponential Decay}
\begin{equation}
\eta_t = \eta_0 \cdot \gamma^t
\end{equation}

\subsubsection{Cosine Annealing}
\begin{equation}
\eta_t = \eta_{min} + \frac{1}{2}(\eta_0 - \eta_{min})(1 + \cos(\frac{\pi t}{T}))
\end{equation}

\subsubsection{Linear Decay}
\begin{equation}
\eta_t = \eta_0 - \frac{\eta_0 - \eta_{min}}{T} \cdot t
\end{equation}

\subsection{Error Cause Analysis}

\subsubsection{Highway $\rightarrow$ River Confusion}
\begin{itemize}
    \item \textbf{Visual Similarity}: Both highways and rivers appear as elongated linear structures in satellite imagery
    \item \textbf{Color Confusion}: Some highways have dark asphalt surfaces that appear similar to water
    \item \textbf{Loss of Spatial Context}: The MLP flattens images, losing spatial structural information that could help distinguish between these classes
\end{itemize}

\subsubsection{AnnualCrop $\rightarrow$ PermanentCrop Confusion}
\begin{itemize}
    \item \textbf{Similar Texture}: Both crop types exhibit similar regular patterns in satellite images
    \item \textbf{Seasonal Variation}: Images captured in different seasons may show different appearances
    \item \textbf{Spectral Overlap}: The spectral signatures of different crop types can be similar
\end{itemize}

\subsubsection{HerbaceousVegetation $\rightarrow$ Pasture Confusion}
\begin{itemize}
    \item \textbf{Vegetation Overlap}: Both classes represent vegetation cover with similar colors and textures
    \item \textbf{Fuzzy Boundaries}: The class definitions have some inherent overlap
\end{itemize}
